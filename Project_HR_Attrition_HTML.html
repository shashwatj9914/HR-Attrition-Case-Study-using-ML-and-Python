def labelling(avg_sat):
    if avg_sat >=0.80:
        return 'A'
    if avg_sat >=0.50:
        return 'B'
    if avg_sat >=0.30:
        return 'C'
    if avg_sat >=0.20:
        return 'D'
    if avg_sat >=0.10:
        return 'E'
    return "F"
